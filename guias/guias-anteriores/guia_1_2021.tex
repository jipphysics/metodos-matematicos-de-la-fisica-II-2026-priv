\begin{equation}\label{T}

\hat e^i)\Bigr),\qquad \forall~i,j\\
A(\hat e^i, \hat e^j) =& \frac{1}{2}\Bigl(T(\hat e^i, \hat e^j) - T(\hat e^j,
\hat e^i)\Bigr),\qquad \forall~i,j
\end{split}\]
\begin{enumerate}
\item Escriba $S$ y $A$ explícitamente en la base producto (como en \eqref{T}). 
\item Calcule además $S+A$, >resulta igual a $T$?, >debería?. Explique.
\end{enumerate}
