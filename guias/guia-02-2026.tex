\section*{Parte I: Problemas Centrales (para desarrollar en clase)}


\prob Sea $\mat{A} = \begin{pmatrix} 1 & 2 \\ 0 & 2 \end{pmatrix}$.
\begin{enumerate}
\item Halle los autovalores y los correspondientes autovectores a derecha e izquierda.
\item Construya la matriz~$\mat{U}$ que tiene a los autovectores a izquierda como filas y la martiz~$\mat{V}$ que tiene a los autovectores a derecha como columnas, y verifique que $\mat{U}\mat{V}=\mat{I}$ y que $\mat{U}\mat{A}\mat{V}$ es diagonal.
\item Halle la norma espectral de~$\mat{A}$.
\item Muestre que~$\mat{A}$ no es normal.
\end{enumerate}

\prob Diagonalice la matriz $\mat{A}$, y escriba el vector $\mat{x}$ en la base donde $\mat{A}$ es diagonal, para
\[
\mat{A} = \begin{pmatrix} 
0 & -i & 0 & 0 & 0 \\ 
i & 0 & 0 & 0 & 0 \\ 
0 & 0 & 3 & 0 & 0 \\ 
0 & 0 & 0 & 1 & -i \\ 
0 & 0 & 0 & i & 1 
\end{pmatrix}, \qquad 
\mat{x} = \begin{pmatrix} 1 \\ a \\ i \\ b \\ -1 \end{pmatrix}.
\]

\prob Halle la forma canónica de Jordan de la matriz
\[
\mat{A} = \begin{pmatrix} 
2 & 1 & 0 & -1 & 3 & 1 \\ 
0 & 2 & 0 & 0 & 1 & -3 \\ 
0 & 0 & 2 & 1 & -1 & 0 \\ 
0 & 0 & 0 & 2 & 1 & -2 \\ 
0 & 0 & 0 & 0 & 2 & 1 \\ 
0 & 0 & 0 & 0 & 0 & 2 
\end{pmatrix}.
\]

\prob Suponga que $\mat{A}$ tiene una descomposición de Schur $\mat{A}=\mat{UTU}^{\dagger}$.
\begin{enumerate}
\item Muestre que para cualquier polinomio $p$ vale que $p(\mat{A})=\mat{U}p(\mat{T})\mat{U}^{\dagger}$.
\item Discuta porqué calcular $p(\mat{T})$ es más fácil que calcular $p(\mat{A})$. ?`Cómo luce $p(\mat{T})$?
\item Si $\mat{A}$ es normal, es decir si $\mat{A}^{\dagger}\mat{A}=\mat{AA}^{\dagger}$, muestre que $\mat{T}$ es diagonal y que $\mat{A}$ es unitariamente diagonalizable. Muestre también que una matriz puede ser diagonalizable sin ser normal. 
\end{enumerate}

\prob Demuestre las siguientes identidades vectoriales en coordenadas cartesianas:
\begin{enumerate}
\item $\nabla(\phi\psi) = \phi\nabla\psi + \psi\nabla\phi$;
\item $\nabla \times (\phi\vec{F}) = (\nabla\phi) \times \vec{F} + \phi(\nabla \times \vec{F})$;
\item $\nabla \cdot (\nabla \times \vec{F}) = 0$;
\item $\nabla \times (\nabla \times \vec{F}) = \nabla(\nabla \cdot \vec{F}) - \nabla^2 \vec{F}$.
\end{enumerate}

\prob Considere el espacio $T^0_2$ de tensores dos veces covariantes sobre $V$ ($\dim(V)=3$), y los subespacios $S$ de tensores simétricos y $A$ de tensores antisimétricos. 
\begin{enumerate}
\item Proponga una base para $T^0_2$ en términos de la base $\{\svec{e}^i\}$ de $V^*$, y escriba explícitamente sus elementos.
\item Escriba explícitamente los elementos de las correspondientes bases de $S$ y $A$. ?`Cuántos elementos tiene cada una?
\end{enumerate}

\prob Muestre que en coordenadas arbitrarias
\begin{equation*}
\epsilon^{ijk}\epsilon_{klm} = 
\delta\ud{i}{l}\delta\ud{j}{m} - \delta\ud{i}{m}\delta\ud{j}{l} \, .
\end{equation*}
Ayuda: muestre primero que en coordenadas Cartesianas ortonormales, donde los \'indices pueden subirse y bajarse arbitrariamente, $\epsilon_{ijk}\epsilon_{klm} = \delta_{il}\delta_{jm} - \delta_{im}\delta_{jl}$, usando todas las simetr\'{\i}as disponibles para trabajar menos; luego suba los \'{\i}ndices que correspondan, transforme a coordenadas arbitrarias y verifique la cancelaci\'on de pesos en el lado izquierdo.

\prob Determine el caracter tensorial (rango, tipo y peso) de las siguientes entidades:
\begin{enumerate}
\item La densidad de carga el\'ectrica de un cuerpo extenso, tomando en cuenta que la carga el\'ectrica total es un escalar.
\item La densidad de masa de un cuerpo extenso, tomando en cuenta que la masa total es un escalar.
\item El momento cuadrupolar el\'ectrico de un cuerpo puntual, tomando en cuenta que la energ\'{\i}a electrost\'atica de un cuadrupolo es un escalar, y que el campo el\'ectrico es un vector (determine su tipo).
\item El ``tensor de inercia'' de un cuerpo r\'{\i}gido, tomando en cuenta que la energ\'{\i}a cin\'etica de rotaci\'on es un escalar, y que la velocidad angular es un {\em pseudovector} (densidad vectorial).
\end{enumerate}

\prob Considere la superficie $S$ definida por $z = \cos x \cos y$ en $\mathbb{R}^3$.
\begin{enumerate}
\item Parametrice $S$ y construya la métrica inducida.
\item Calcule el diferencial de área escalar y el área del parche $x,y \in [-\pi,\pi]$.
\item Calcule la longitud de la curva $x=y$, $z = \cos x \cos y$ desde $x=-\pi$ hasta $x=\pi$.
\end{enumerate}

\prob En coordenadas esféricas $(r,\theta,\phi)$:
\begin{enumerate}
\item Construya la métrica, los factores de escala y el diferencial de volumen.
\item Calcule los símbolos de Christoffel (conexión de Levi-Civita).
\item Escriba las expresiones del gradiente, divergencia, rotor y laplaciano en términos de componentes físicas.
\end{enumerate}

\prob El potencial electrostático de un dipolo $\vec{p} = p\hat{z}$ colocado en el origen, es $\varphi = \dfrac{p}{4\pi\epsilon_0} \dfrac{\cos\theta}{r^2}$ en coordenadas esféricas. Calcule $\vec{E} = -\nabla\varphi$, $\nabla \times \vec{E}$ y $\nabla \cdot \vec{E}$ en esféricas.

\prob Considere coordenadas esferoidales proladas
\[
x = a \sinh\mu \sin\nu \cos\phi,\quad y = a \sinh\mu \sin\nu \sin\phi,\quad z = a \cosh\mu \cos\nu.
\]
\begin{enumerate}
\item Muestre que son ortogonales y describa las superficies coordenadas.
\item Calcule los factores de escala.
\item Obtenga las fórmulas del gradiente, divergencia, rotor y laplaciano en componentes f\'{\i}sicas.
\end{enumerate}



\section*{Parte II: Problemas Propuestos (para profundización y repaso)}

\prob Verifique que en coordenadas esféricas se cumple $\nabla(\phi\psi)=\phi\nabla\psi+\psi\nabla\phi$ y $\nabla\times(\phi\vec{F})=(\nabla\phi)\times\vec{F}+\phi(\nabla\times\vec{F})$.

\prob Demuestre que si $A_{ij}$ es antisimétrico y $S_{ij}$ es simétrico, entonces $A_{ij}S^{ij}=0$. Muestre además que todo tensor de rango 2 se descompone en suma de su parte simétrica y antisimétrica.

\prob Muestre que el operador identidad ${\cal I}:V\to V$ corresponde al tensor $I=\delta^i_{\,j}\,\mathbf{e}_i\otimes\mathbf{e}^j$ y calcule $\operatorname{Tr}(I)$.

\prob Dado un tensor $T=T_{ijk}\,\svec{e}^i\otimes\svec{e}^j\otimes\svec{e}^k$, ¿es cierto que $T_{ijk}=T_{\{ijk\}}+T_{[ijk]}$? Justifique.

\prob Demuestre que el producto tensorial de dos densidades tensoriales de pesos $p_1$ y $p_2$ es una densidad tensorial de peso $p_1+p_2$.

\prob El potencial vector de un dipolo magnético $\vec{m}=m\hat{z}$ ubicado en el origen es $\vec{A}=\dfrac{\mu_0 m}{4\pi}\dfrac{1}{r^3}\hat{\phi}$ en coordenadas esféricas. Calcule $\vec{B}=\nabla\times\vec{A}$ y $\nabla\cdot\vec{B}$ en esféricas.

\prob Encuentre los autovalores y autovectores de las matrices de Pauli
\[
\sigma_x=\begin{pmatrix}0&1\\1&0\end{pmatrix},\quad \sigma_y=\begin{pmatrix}0&-i\\i&0\end{pmatrix},\quad \sigma_z=\begin{pmatrix}1&0\\0&-1\end{pmatrix}.
\]

\prob Calcule la métrica inducida en la esfera $r=R$ a partir de la métrica euclídea en $\mathbb{R}^3$ en coordenadas esféricas.

\prob Demuestre que $\nabla \cdot (\vec{F} \times \vec{G}) = (\nabla \times \vec{F}) \cdot \vec{G} - \vec{F} \cdot (\nabla \times \vec{G})$ en coordenadas cartesianas y verifíquelo en esféricas.

\prob Halle la forma canónica de Jordan de $\mat{A}=\begin{pmatrix}1&-1&2\\2&-2&0\\2&-1&3\end{pmatrix}$.
