\section*{Parte I: Problemas Centrales (para desarrollar en clase)}


\prob Determine cuáles de las siguientes funciones están en $L^2(0,\infty)$; para aquellas que lo estén, calcule su norma.\\
\begin{center}
a) $e^{-x}$.\qquad
b) $\operatorname{sen} x$.\qquad
c) $\displaystyle \frac{1}{1+x}$.\qquad
d) $\displaystyle \frac{1}{\sqrt{x}}$.
\end{center}

\prob Determine el límite en $L^2$ de cada una de las siguientes sucesiones de funciones, si existe.\\
\begin{center}
a) $\sqrt{x}$, $0\le x\le 1$.\qquad
b) $\displaystyle \begin{cases}
nx, & 0 \le x < 1/n,\\
1, & 1/n \le x \le 1.
\end{cases}$\qquad
c) $n x (1-x)^n$, $0\le x\le 1$.
\end{center}

\prob Muestre que las funciones $\{1, \cos(nx), \operatorname{sen}(nx),\; n=1,2,\dots\}$ son ortogonales dos a dos en $L^2[-\pi,\pi]$. Ídem para las funciones $\{e^{inx},\; n\in\mathbb{Z}\}$.

\prob (Desigualdad de Hölder) Sea la norma $L^p$:
\[
\|u\|_{L^p} = \left( \int_a^b |u(x)|^p \,dx \right)^{1/p}, \qquad p\ge 1.
\]
Pruebe que vale la desigualdad
\[
\|uv\|_{L^s} \le \|u\|_{L^p} \|v\|_{L^q}, \qquad \text{para } s^{-1}=p^{-1}+q^{-1}.
\]
Sugerencia: Utilice la desigualdad de Hölder para la función $(uv)^s$.

\prob Pruebe que si $T:X\to Y$ es lineal ($X$ e $Y$ son espacios de Banach) entonces $T$ es continua en todo punto de $X$ si y solo si $T$ es continua en cero.

\prob Sea $H$ un espacio de Hilbert separable y $\{\hat{e}_1,\hat{e}_2,\dots\}$ una base ortonormal. Pruebe que si $x\in H$,
\[
\text{a) } x = \sum_{j=1}^{\infty} \langle \hat{e}_j, x \rangle \hat{e}_j, \qquad
\text{b) Identidad de Parseval: } \|x\|^2 = \sum_{j=1}^{\infty} |\langle \hat{e}_j, x \rangle|^2.
\]

\prob Considere la función de prueba $\varphi \in C_0^\infty(\mathbb{R})$ definida como
\[
\varphi(x) = \begin{cases}
e^{-\frac{1}{1-x^2}}, & |x| < 1,\\
0, & |x| \geq 1.
\end{cases}
\]
Muestre que, efectivamente, $\varphi$ y todas sus derivadas son continuas en $\mathbb{R}$.

\prob Considere la sucesión de funciones en $C^\infty(\mathbb{R})$
\[
f_n = \frac{n}{\sqrt{\pi}} e^{-n^2 x^2}.
\]
Muestre que
\[
f_n \xrightarrow{n \to \infty} \delta.
\]
¿A qué converge $f'_n$? ¿Y $f_n^{(k)}$? (Ayuda: note que toda función de prueba en $C_0^\infty(\mathbb{R})$ o $\mathcal{S}(\mathbb{R})$ es entera, y por consiguiente tiene un desarrollo de Taylor con radio de convergencia infinito.)

\prob Muestre que
\[
\int_a^b f(x)\,\delta(g(x))\,dx = \frac{f(x_0)}{|g'(x_0)|},
\]
asumiendo que la ecuación $g(x)=0$ posee una sola raíz en el intervalo $a<x<b$.

\prob Sea $f\in L^1(\mathbb{R})$ y sea $T_f$ la funcional definida como
\[
T_f(\varphi) = \int_0^{\infty} f(x)\varphi(x)\,dx,
\]
para toda $\varphi\in\mathcal{D}(\mathbb{R})$. Muestre que $T_f$ es una distribución.

\prob Pensando a la delta de Dirac como una función generalizada, pruebe las siguientes propiedades:
\begin{enumerate}
\item $\displaystyle \int_{-a}^{b} \delta(t)\,dt = 1$, para $a,b>0$.
\item $\displaystyle \int \delta(t-a)\,dt = 1$, si el rango de integración incluye $t=a$.
\item $\delta(t) = \delta(-t)$.
\item $\displaystyle \delta(at) = \frac{1}{|a|}\delta(t)$.
\item $t^n\delta(t)=0$, con $n>0$.
\item $\displaystyle \delta(h(t)) = \sum_i \frac{\delta(t-t_i)}{|h'(t_i)|}$, donde $t_i$ son todos los valores de $t$ para los cuales $h(t)=0$.
\item $\displaystyle h(t)\delta'(t-a) = h(a)\delta'(t-a) - h'(a)\delta(t-a)$.
\end{enumerate}

\prob Sean $f:\mathbb{R}\to\mathbb{R}$, $\varphi$ y $\psi$ dos funciones de prueba. Considere la distribución definida como
\[
T_f(\phi) = \int_{-\infty}^{\infty} f(x)\phi(x)\,dx.
\]
\begin{enumerate}
\item Muestre que $\operatorname{sen}(ax)\,\delta'(x) = -a\delta(x)$, con $a\in\mathbb{R}$.
\item Demuestre que $(\varphi\psi)' = \varphi'\psi + \varphi\psi'$ en el sentido distribucional.
\item Utilizando los incisos anteriores, muestre que
\[
\frac{d^2}{dx^2}\bigl[H(x)\operatorname{sen}(ax)\bigr] = a\delta(x) - H(x)\,a^2\operatorname{sen}(ax),
\]
donde $H(x)$ es la función escalón de Heaviside.
\end{enumerate}



\section*{Parte II: Problemas Propuestos (para profundización y repaso)}


\prob Sea la norma $l^p$:
\[
\|u\|_{l^p} = \left( \sum_{j=0}^{\infty} |u^j|^p \right)^{1/p}, \qquad p\ge 1.
\]
\begin{enumerate}
\item Pruebe que está bien definida.
\item Pruebe que el subconjunto de sucesiones con norma $l^p$ finita está contenido en el subconjunto de las sucesiones con norma $l^q$ finita si $p<q$.
\end{enumerate}

\prob Pruebe que $\|x\| = \sqrt{\langle x,x\rangle}$ satisface la desigualdad triangular. A tal fin utilice la desigualdad de Schwarz.

\prob Considere el espacio de sucesiones infinitas de números complejos con norma $l^2$ finita. Sea $\hat{e}_i\in l^2$ la sucesión que tiene un $1$ en el $i$-ésimo lugar y cero en todos los restantes. Pruebe que definiendo estos elementos para $i=1,2,\dots$, el conjunto $B=\{\hat{e}_1,\hat{e}_2,\dots\}\subset l^2$ es una base ortonormal de $l^2$.

\prob Sea $H$ un espacio de Hilbert y $\{\hat{e}_1,\hat{e}_2,\dots,\hat{e}_N\}$ un conjunto ortonormal (no necesariamente una base). Pruebe que
\begin{enumerate}
\item Identidad de Pitágoras: si $x\in H$,
\[
\|x\|^2 = \sum_{j=1}^{N} |\langle \hat{e}_j,x\rangle|^2 + \Bigl\|x - \sum_{j=1}^{N} \langle \hat{e}_j,x\rangle \hat{e}_j\Bigr\|^2.
\]
\item Desigualdad de Bessel: $\displaystyle \sum_{j=1}^{N} |\langle \hat{e}_j,x\rangle|^2 \le \|x\|^2$.
\end{enumerate}

\prob Muestre que $C_0^\infty(\mathbb{R})$ no s\'olo es un espacio vectorial, sino además un álgebra con el producto usual de funciones. Halle el soporte de $fg$ en términos de los soportes de $f$ y $g$.

\prob Demuestre que $h(x)\in C^\infty(\mathbb{R})$, donde
\[
h(x) = \begin{cases}
0, & \text{si } x\le 0,\\
\exp(-1/x^2), & \text{si } x>0.
\end{cases}
\]

\prob Muestre que en el caso de tener un espacio de Banach con la noción de convergencia dada por la norma, la definición de continuidad de una funcional coincide con la definición usual $\varepsilon$-$\delta$. Muestre además que, como con cualquier noción de continuidad, para ser verificada en el caso de una funcional lineal, es suficiente verificarla en un solo punto y eso garantiza la continuidad en todos lados.

\prob Demuestre que la derivada de una distribución es lineal.

\prob Dada
\[
h(x) = \begin{cases}
\exp(x+1), & \text{si } x<-1,\\
x^4, & \text{si } -1\le x\le 0,\\
\operatorname{sen}(x), & \text{si } 0<x<\pi/2,\\
1, & \text{si } \pi/2\le x,
\end{cases}
\]
calcule las dos primeras derivadas de $h$ en el sentido distribucional.
