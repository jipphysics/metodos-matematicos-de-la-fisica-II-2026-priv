\section{Metric space}

A set $X$ is called a {\em metric space} if it is equiped with a notion of {\em metric distance}, i.e. a function $d:X\times X\to \RR$ such that, for any three {\em points} $x$, $y$ and $z$ in $X$, it holds:
\begin{enumerate}
\item $d(x,y) \geq 0$, i.e. it is {\em non-negative}.
\item $d(x,y) = 0$ implies $x=y$, i.e. it is {\em positive definite}.
\item $d(x,y) = d(y,x)$, i.e. it is {\em symmetric}.
\item $d(x,z) \leq  d(x,y) + d(y,z)$, i.e. it satisfies the {\em triangular inequality}.
\end{enumerate}

{\em Example:} The set $\RR^n$ with the euclidean distance $d(x,y) = \sqrt{\sum_{i=1}^n (x_i-y_i)^2}$ is a metric space.

\subsection{Open ball}

The set
$$
B_r(x)
=
\{y\in \RR^n: d(x,y)<r\}
$$
is called the {\em open ball} of $\RR^n$ centered at $x$ and of radious $r>0$.

\subsection{Open set}

A subset $U$ of $\RR^n$ is {\em open} if for every $x\in U$ there exists an open ball $B_r(x) \subset U$.

\subsection{Complement set}

Consider two subsets $X$ and $Y$ of $\RR^n$ such that $X\subset Y$.
The {\em complement} $X\backslash Y$ of $X$ in $Y$ is the subset of $\RR^n$ given by
$$
X\backslash Y := \{y\in Y:y\notin X\}.
$$
In particular, we can think of the complement $X\backslash \RR^n$.

\subsection{Closed set}

A subset $F$ of $\RR^n$ is {\em closed} if its complement $F\backslash \RR^n$ is open.

\subsection{
Limit or accumulation point
\label{sec:limit_point_metric}
}

A point $x$ is a {\em limit point} (or {\em accumulation point}) of a subset $A$ of $\RR^n$ if, for every $r>0$, it holds that
$$
(B_r(x)\backslash \{x\})\cap A \neq \emptyset.
$$
In other words, there are points in $A$ that are arbitrarily close to $x$.

{\em Example:}
The points $0$ and $1$ are limit points of the open interval $(0,1)$.

\subsection{Closure set}

The {\em closure} $\overline{A}$ of a subset $A$ of $\RR^n$ is defined by
$$
\overline{A}
=
A \cup \{\mbox{all limiting points of $A$}\}.
$$

{\em Examples:} the closed interval $[0,1]$ is the closure of itself and the subsets $(0,1)$, $[0,1)$, $(0,1]$ and $(0,1/2)\cup (1/2,1)$.

\subsection{Bounded set}

A subset $A$ of $\RR^n$ is {\em bounded} if there exists a finite $r>0$ such that $B_r(0)\cap A=A$.

\subsection{Compact set}

A subset $K$ of $\RR^n$ is {\em compact} if $K$ is closed and bounded.

\section{Topological space}

Previous definitions of open and closed sets depend on the existence of a metric distance.
It is possible to generalize the formalism using a weaker, i.e. more general, requirement.
Namely, the notions of open and closed sets can be defined when the set is equipped with a notion of topology, which provides a weaker notion of space, than that of a metric.

%\subsection{Topological space}

A {\em topological space} is a set $X$ equipped with a collection $T$ of subsets of $X$ called {\em open} sets that satisfy the following properties:
\begin{enumerate}
\item
$X$ and $\emptyset$ belong to $T$.
\item
Arbitrary unions of open sets are open.
Formally, for any $R\subset T$, then 
$$
\bigg( \bigcup_{U\in R}\, U \bigg) \in T.
$$
\item
Finite intersections of open sets are open.
Formally, for any $R\subset T$ such that $|R|\in \NN$, then
$$
\bigg( \bigcap_{U\in R}\, U \bigg) \in T.
$$
\end{enumerate}
In other words, the notion of openness is a primite definition.

The notions of {\em complement} set and {\em closed} set are analogous to those previously defined.
Namely, the complement of a subset $S$ of $X$ is the subset $S\backslash X := \{x\in X:x\notin S\}$ of $X$, and a subset $Y$ of $X$ is closed if its complement $Y\backslash X$ is open.

\subsection{Closure set}

The {\em closure} $\overline{A}$ of a subset $A$ of $X$ is defined as the intersection of all closed subsets of $X$ that contain $A$.
Formally,
$$
\overline{A}
=
\bigcap_{F:F\subset X,\,A\subset F,\,F\, \text{is closed}}\,F.
$$

\subsection{Limit point}

A point $x$ is a {\em limit point} (or {\em accumulation point}) of a subset $A$ of $X$ if, for each open subset $U$ of $X$ such that $x\in U$, then
$$
(U\backslash \{x\})\cap A \neq \emptyset.
$$
This is the purely topological generalization of the notion defined in section~\ref{sec:limit_point_metric}.

\subsection{Open cover}

A family of open sets $\{U_i:i\in I\}\subset\mathcal T$ is called an {\em open cover} of $K$ if
$$
K\subset \bigcup_{i\in I} U_i.
$$

\subsection{Compact set}

A subset $K$ of $X$ is {\em compact} if for every open cover $\{U_i:i\in I\}$ of $K$, there exists a finite subset
$$
\{i_1,\dots,i_n\}\subset I
$$
such that
$$
K\subset
U_{i_1}\cup\cdots\cup U_{i_n}.
$$
The finite family
$$
{U_{i_1},\dots,U_{i_n}}
$$
is called a {\em finite subcover}.

{\em Example:}
The family
$$
U_n=\left(\frac1n,1\right),
\qquad n\in\mathbb N,
$$
is an open cover of $(0,1)$ because
$$
(0,1)
=
\bigcup_{n=1}^\infty
\left(\frac1n,1\right),
$$
but no finite subcollection of it covers $(0,1)$.
Namely, if we choose finitely many:
$$
U_{n_1},\dots,U_{n_k}.
$$
then 
$$
U_{n_1}\cup\cdots\cup U_{n_k}
=
\left(\frac1N,1\right).
$$
with $N=\max(n_1,\dots,n_k)$, missing all points in $(0,1/N]$.
Hence, the finite union does not cover $(0,1)$.
Thus $(0,1)$ admits an open cover with no finite subcover, and therefore it is not compact.

\section{Test functions}

\subsection{Support of a function}

The {\em support} of a function $\varphi : \RR^n\to \RR$, is the subset $K$ of $\RR^n$ given by:
$$
\supp(\varphi)
=
\overline{
\left\{
x\in \RR^n
:
\varphi(x) \neq 0
\right\}
}.
$$
In other words, it is the closure of the complement of the kernel of $\varphi$.

\subsection{Smooth function}

Let $C^{\infty}(\RR^n)$ denote the set of {\em smooth functions} (i.e. of infinitelly differentiable functions) from $\RR^n$ to $\RR$.

\subsection{Space of compactly supported smooth test functions}

The (vector) {\em space of smooth compactly supported functions} $C_c^{\infty}(\RR^n)=\mathcal{D}(\RR^n)$ is such that, if $\varphi\in C_c^{\infty}(\RR^n)$, then:
\begin{enumerate}
\item
$\varphi : \RR^n \to \RR$.

\item
$\varphi \in C^{\infty}(\RR^n)$, i.e. it is infinitely differentiable.

\item
The support $\supp(\varphi)$ is compact.
\end{enumerate}
The members of $C_c^{\infty}(\RR^n)$ are called {\em compactly supported smooth test functions}.

{\em Excercise:} 
Show that the linear combination and the product of compactly supported smooth test functions are also compactly supported smooth test functions.

\subsection{Schwartz space of functions}

The {\em Schwartz (vector) space of functions} $\mathcal{S}(\RR^n)$ is such that, if $\varphi\in \mathcal{S}(\RR^n)$, then:
\begin{enumerate}
\item
$\varphi : \RR^n \to \RR$.

\item
$\varphi \in C^{\infty}(\RR^n)$, i.e. it is infinitely differentiable.

\item For every $\alpha_1,...,\alpha_n,\beta_1,...,\beta_n\in \{0,1,2,...\}$, it holds that
$$
\supr{x\in \RR^n}\,\left|x_1^{\alpha_1}...x_n^{\alpha_n}\partial_1^{\beta_1}...\partial_n^{\beta_n}\varphi(x)\right|<\infty.
$$
In other words, all the derivatives of $\varphi$ multiplied by arbitrary powers of the components of $x$ are bounded in $\RR^n$.
\end{enumerate}
The members of $\mathcal{S}(\RR^n)$ are called {\em Schwartz test functions}.

{\em Excercise:} 
Show that the linear combination and the product of Schwartz test functions are also Schwartz test functions.
